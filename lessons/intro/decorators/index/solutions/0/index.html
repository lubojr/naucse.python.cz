def co_se_deje(func):
    def nahradni_funkce(x):
        print(f"Voláme {func.__name__}({x})")
        vysledek = func(x)
        print(f"Výsledek {func.__name__}({x}) = {vysledek}")
        return vysledek

    return nahradni_funkce
